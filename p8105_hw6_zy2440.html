  y = log(x[[2]][1]*x[[2]][2])
  y
}

rsq <- function(x){
  y=broom::glance(x)[[1]]
}

boot_straps = 
  weather_df %>% 
  modelr::bootstrap(n = 5000) %>%
  mutate(
    models = map(strap, ~lm(tmax ~ tmin, data = .x)),
    results = map(models, broom::tidy),
    r2 = map(models,rsq),
    log= map(results,logtimes)) %>%
  select(r2,log,.id)
boot_straps <- as.data.frame(boot_straps) %>%
